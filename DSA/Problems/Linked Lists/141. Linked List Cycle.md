# Solution

## [[Hash Set]] - O(n) time - O(n) space

```cpp
class Solution {
public:
    bool hasCycle(ListNode *head) {
        unordered_set<ListNode*> list;
        while(head) {
            if (list.count(head)) return true;
            else {
                list.insert(head);
                head = head->next;
            }
        }
        return false;
    }
};
```
